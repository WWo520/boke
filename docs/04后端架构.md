## 三、核心组件

### 3.1 应用入口（`src/index.js`）

**职责**：初始化 Express 应用、配置中间件、注册路由、启动服务

```javascript
const express = require('express');
const cors = require('cors');
const multer = require('multer');
const { initDb } = require('./db');

const app = express();

app.use(cors());
app.use(express.json({ limit: '10mb' }));
app.use('/uploads', express.static('uploads'));

const upload = multer({ dest: 'uploads/' });

app.post('/api/upload', upload.single('file'), (req, res) => {
  res.json({ url: `/uploads/${req.file.filename}` });
});

app.use('/api/auth', require('./routes/auth'));
app.use('/api/posts', require('./routes/posts'));
app.use('/api/users', require('./routes/users'));
app.use('/api/comments', require('./routes/comments'));
app.use('/api/categories', require('./routes/categories'));
app.use('/api/admin', require('./routes/admin'));

app.get('*', (req, res) => {
  res.json({ message: 'API is running' });
});

initDb().then(() => {
  app.listen(3333, () => {
    console.log('Server running on port 3333');
  });
});
```

### 3.2 数据库模块（`src/db.js`）

**职责**：数据库连接池管理、SQL 执行封装

```javascript
const { Pool } = require('pg');

const pool = new Pool({
  user: 'postgres',
  host: 'localhost',
  database: 'moke_blog',
  password: 'postgres',
  port: 5432,
});

async function initDb() {
  await pool.query(`CREATE TABLE IF NOT EXISTS users (...)`);
  await pool.query(`CREATE TABLE IF NOT EXISTS posts (...)`);
  await pool.query(`CREATE TABLE IF NOT EXISTS categories (...)`);
  await pool.query(`CREATE TABLE IF NOT EXISTS post_tags (...)`);
  await pool.query(`CREATE TABLE IF NOT EXISTS comments (...)`);
  await pool.query(`CREATE TABLE IF NOT EXISTS post_likes (...)`);
  await pool.query(`CREATE TABLE IF NOT EXISTS post_favorites (...)`);
  
  await require('./seedData')();
}

async function queryOne(sql, params = []) {
  const result = await pool.query(sql, params);
  return result.rows[0] ? camelizeRow(result.rows[0]) : null;
}

async function queryAll(sql, params = []) {
  const result = await pool.query(sql, params);
  return result.rows.map(camelizeRow);
}

async function runSql(sql, params = []) {
  const result = await pool.query(sql, params);
  return { changes: result.rowCount };
}

async function insertSql(sql, params = []) {
  const result = await pool.query(sql, params);
  return result.rows[0].id;
}

function camelizeRow(row) {
  const result = {};
  for (const key of Object.keys(row)) {
    result[key.replace(/(_\w)/g, m => m[1].toUpperCase())] = row[key];
  }
  return result;
}

module.exports = {
  pool,
  initDb,
  queryOne,
  queryAll,
  runSql,
  insertSql,
};
```

### 3.3 认证中间件（`src/authMiddleware.js`）

**职责**：验证 JWT 令牌、提取用户信息

```javascript
const jwt = require('jsonwebtoken');
const { queryOne } = require('./db');

const JWT_SECRET = process.env.JWT_SECRET || 'secret';

async function authMiddleware(req, res, next) {
  const authHeader = req.headers.authorization;
  
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    return res.status(401).json({ error: { code: 'UNAUTHORIZED', message: '未授权' } });
  }
  
  const token = authHeader.split(' ')[1];
  
  try {
    const decoded = jwt.verify(token, JWT_SECRET);
    const user = await queryOne('SELECT id, name, email, avatar, role FROM users WHERE id = $1', [decoded.userId]);
    
    if (!user) {
      return res.status(401).json({ error: { code: 'UNAUTHORIZED', message: '用户不存在，请重新登录' } });
    }
    
    req.user = user;
    next();
  } catch (err) {
    return res.status(401).json({ error: { code: 'UNAUTHORIZED', message: '令牌无效' } });
  }
}

module.exports = authMiddleware;
```

### 3.4 管理员中间件（`src/adminMiddleware.js`）

**职责**：验证用户是否为管理员

```javascript
function adminMiddleware(req, res, next) {
  if (!req.user || req.user.role !== 'admin') {
    return res.status(403).json({ error: { code: 'FORBIDDEN', message: '无权限访问' } });
  }
  next();
}

module.exports = adminMiddleware;
```

## 四、路由模块

### 4.1 认证路由（`routes/auth.js`）

| 路由 | 方法 | 中间件 | 说明 |
|------|------|--------|------|
| `/api/auth/login` | POST | - | 用户登录 |
| `/api/auth/register` | POST | - | 用户注册 |
| `/api/auth/me` | GET | authMiddleware | 获取当前用户 |

**登录流程**：
1. 接收 email、password
2. 查询用户（按 email）
3. bcrypt.verify() 验证密码
4. 生成 JWT（有效期 7 天）
5. 返回 token 和用户信息

**注册流程**：
1. 接收 name、email、password
2. 验证邮箱是否已存在
3. bcrypt.hash() 加密密码
4. 插入 users 表
5. 返回用户信息

### 4.2 文章路由（`routes/posts.js`）

| 路由 | 方法 | 中间件 | 说明 |
|------|------|--------|------|
| `/api/posts` | GET | - | 获取文章列表（分页） |
| `/api/posts/:slug` | GET | - | 获取文章详情 |
| `/api/posts` | POST | authMiddleware | 创建文章 |
| `/api/posts/:slug` | PUT | authMiddleware | 更新文章 |

**文章列表查询**：

**文章创建**：
- 验证必填字段（标题必须有，发布状态需完整字段）
- 生成 slug（标题转小写 + 时间戳）
- 插入 posts 表
- 插入 post_tags 关联表

## 五、认证与权限

### 5.1 JWT 配置

| 配置项 | 值 |
|--------|------|
| 密钥 | `process.env.JWT_SECRET` 或 'secret' |
| 有效期 | 7 天（`7d`） |
| 算法 | HS256 |

### 5.2 用户角色

| 角色 | 权限 |
|------|------|
| `user` | 浏览、评论、发布文章、管理个人资料 |
| `admin` | 所有权限 + 管理面板 |

### 5.3 权限控制

- **公开接口**：无需认证
- **用户接口**：需登录（authMiddleware）
- **管理接口**：需管理员权限（adminMiddleware）
- **资源归属**：用户只能修改自己的文章/评论

## 六、文件上传

### 6.1 上传配置

```javascript
const multer = require('multer');

const storage = multer.diskStorage({
  destination: 'uploads/',
  filename: (req, file, cb) => {
    const ext = file.originalname.split('.').pop();
    cb(null, `${Date.now()}-${Math.random().toString(36).substr(2, 9)}.${ext}`);
  },
});

const upload = multer({ 
  storage,
  limits: { fileSize: 10 * 1024 * 1024 },
});
```

### 6.2 上传流程

1. 前端压缩图片（Canvas）
2. 调用 POST `/api/upload`
3. multer 保存到 `uploads/` 目录
4. 返回文件 URL `/uploads/filename`

## 八、安全考虑

### 8.1 SQL 注入防护

- 使用参数化查询（pg 的参数数组）
- 禁止字符串拼接 SQL

### 8.2 密码安全

- bcryptjs 哈希（10轮）
- 禁止明文存储密码
- 禁止返回密码字段

### 8.3 JWT 安全

- 密钥存储在环境变量
- 令牌有效期限制（7天）
- 过期令牌及时清理

### 8.4 文件上传安全

- 文件名随机化
- 文件大小限制（10MB）
- 禁止执行上传文件

### 8.5 跨域安全

- CORS 配置允许前端域名
- 禁止全局跨域
